\subsubsection{Listas}
\begin{itemize}
  \item Uma \textbf{lista} é uma seqüęncia ordenada de objetos. Ex.: $(1, 2, 3)$ ou ]
        $(1, 2, \Z)$.
  \item Uma lista pode conter \emph{elementos repetidos}.
  \item Duas listas săo iguais se seus elementos săo iguais.
  \item O número de listas de comprimento $k$, cujos elementos săo escolhidos de um 
        conjunto de $n$ elementos possíveis, é igual a:
  \begin{itemize}
    \item $n^k$, caso se permitam repetiçőes;
    \item $(n)_k$, caso năo se permitam repetiçőes, onde 
          $(n)_k = n(n-1)(n-2)\dotsb(n-k+1)$.
  \end{itemize}
  \item Em um torneio estăo sendo dados tręs pręmios de melhor jogador para 100 
        jogadores. Se o mesmo jogador pode receber mais de um pręmio, entăo o número de 
        combinaçőes possíveis é $100^3$. Caso um jogador possa receber apenas um 
        pręmio, o número de combinaçőes é $(100)_3$ -- ou seja -- 
        $(100 \cdot 99 \cdot 98)$.
\end{itemize}

\subsubsection{Teoria dos Conjuntos}
\begin{itemize}
  \item Um \textbf{conjunto} é uma seleçăo de elementos, \emph{sem ordenaçăo} e que
        \emph{năo se repete}.
  \item O número de subconjuntos de $A$ é $2^{|A|}$.
  \item O conjunto potęncia de $A$ se denota por $2^A$ e é o conjunto de todos os 
        subconjuntos de $A$. Por exemplo: $2^{\{1,2,3\}} = \{\emptyset,\{1\},\{2\},
        \{3\}, \{1,2\}, \{1,3\}, \{2,3\}, \{1,2,3\}\}$.
  \item $A \cup B$ é a \emph{uniăo} entre $A$ e $B$ e é o conjunto de todos os 
        elementos que estăo em $A$ e $B$.
  \begin{itemize}
    \item Se $A = \{1,2,3,4\}$ e $B = \{3,4,5,6\}$, entăo $A \cup B = \{1,2,3,4,5,6\}$.
  \end{itemize}
  \item $A \cap B$ é a \emph{intersecçăo} entre $A$ e $B$ e é o conjunto de todos os 
        elementos que estăo tanto em $A$ quanto em $B$.
  \begin{itemize}
    \item Se $A = \{1,2,3,4\}$ e $B = \{3,4,5,6\}$, entăo $A \cap B = \{3,4\}$.
  \end{itemize}
  \item Propriedades dos conjuntos:
  \begin{itemize}
    \item $A \cup B = B \cup A$ (vale também para $\cap$)
    \item $A \cup (B \cup C) = (A \cup B) \cup C$ (vale também para $\cap$)
    \item $A \cup \emptyset = A$ e $A \cap \emptyset = \emptyset$
    \item $A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$
    \item $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$
  \end{itemize}
  \item $|A|+|B|=|A\cup B|+|A\cap B|$
  \item $A$ e $B$ săo \emph{disjuntos} se $A \cap B = \emptyset$.
  \item Se $A_1, A_2, \dotsc, A_n$ săo \emph{disjuntos dois a dois} se 
        $A_i\cap A_j = \emptyset$ para todo $i\neq j$. Isto é, \emph{năo há dois deles 
        que tenham elementos em comum}.
  \item $A - B$ é a \emph{diferença} entre $A$ e $B$ e é o conjunto de todos os 
        elementos que estăo em $A$ e năo estăo em $B$.
  \begin{itemize}
    \item Se $A = \{1,2,3,4\}$ e $B = \{3,4,5,6\}$, entăo $A - B = \{1,2\}$.
  \end{itemize}
  \item $A \Delta B$ é a \emph{diferença simétrica} entre $A$ e $B$ e é o conjunto de 
        todos os elementos que estăo em $A$ mas năo em $B$, ou em $B$ mas năo em $A$.
  \begin{itemize}
    \item Se $A = \{1,2,3,4\}$ e $B = \{3,4,5,6\}$, entăo $A \Delta B = \{1,2,5,6\}$.
  \end{itemize}
  \item $A \Delta B = (A \cup B) - (A \cap B)$
  \item $A \times B$ é o \emph{produto cartesiano} de $A$ e $B$ e é o conjunto de
        todos os pares possíveis com um elemento de $A$ e outro de $B$.
\end{itemize}

\subsubsection{Coeficientes Binomiais}
\begin{itemize}
  \item O número de subconjuntos de $k$ elementos possíveis em um conjunto de $n$
        elementos é denotado por $\binom{n}{k}$.
  $$\binom{n}{k} = \frac{n!}{k!(n-k)!}$$
  \item O triângulo de Pascal facilita o cálculo dos coeficientes binomiais.
\end{itemize}


\subsection{Relaçőes}
\begin{itemize}
  \item Uma \emph{relaçăo} é um conjunto de pares ordenados.
  \item $R$ é uma relaçăo sobre os conjuntos $A$ e $B$ se $R \subseteq A \times B$.
  \item Propriedades das relaçőes:
  \begin{itemize}
    \item $R$ é \textbf{reflexiva} se para todo $x \in A$, entăo $x\;R\;x$. Exemplo: 
          $\geq$.
    \item $R$ é \textbf{anti-reflexiva} se para todo $x \in A$, entăo $x\;R\negate\;x$.
          Exemplo: $>$.
    \item $R$ é \textbf{simétrica} se para todo $x,y \in A$, se $x\;R\;y$ entăo 
          $y\;R\;x$. Exemplo: ``é casado com'', $=$.
    \item $R$ é \textbf{anti-simétrica} se para todo $x,y \in A$, se quando $x\;R\;y$
          e $y\;R\;z$, entăo $x=y$. Exemplo: $\geq$, pois se $x\geq y$ e $y\geq x$,
          entăo $x=y$.
    \item $R$ é \textbf{transitiva} se para todo $x,y,z \in A$, se $x\;R\;y$ e
          $y\;R\;z$, entăo $x\;R\;z$. Exemplo: $>$, pois se $x>y$ e $y>z$, entăo
          $x>z$.
    \begin{itemize}
      \item Contra-exemplo: maternidade \emph{năo} é uma relaçăo transitiva, por se
            Alice é măe de Beatriz, e Beatriz é măe de Clara, entăo Alice \textbf{năo} 
            é măe de Clara.
    \end{itemize}
  \end{itemize}
  \item Uma \textbf{relaçăo de equivalęncia} é aquela que é reflexiva, simétrica e
        transitiva.
  \item Uma \textbf{relaçăo inversa} $R^{-1}$ é definida como 
        $R^{-1}=\{(y,x)|(x,y)\in R\}$. Exemplo: $(\geq)^{-1} = \leq$.
  \item Uma \textbf{classe de equivalęncia} ``particiona'' um conjunto em várias partes,
        onde todos os elementos do conjunto $A$ se relacionam com $[a]$ através da
        relaçăo $R$ quando $[a] = \{x \in R : x\;R\;a\}$.
  \begin{itemize}
    \item O conjunto $\Z$ pode ser divido entre os pares (isto é, entre os
          elementos $x$ onde $x\equiv 0\pmod 2$) e os ímpares (isto é, entre os
          elementos $x$ onde $x\equiv 1\pmod 2$). Isto significa que, para esta relaçăo
          de equivalęncia ($a\pmod 2$), $[0]$ é o conjunto de todos os números pares,
          e $[1]$ o conjunto dos números ímpares.
    \item Assim, $[a]\cap [b] = \emptyset$.
  \end{itemize}
\end{itemize}


\subsection{Funçőes}
\begin{itemize}
  \item Uma \textbf{funçăo} é uma relaçăo $f$ onde se $(a,b) \in f$ e $(a,c) \in f$,
        entăo $b=c$. Caso contrário, a relaçăo năo é uma funçăo.
  \item Isto significa que uma funçăo só pode ter uma saída para cada entrada.
  \item Num gráfico, uma linha nunca pode cruzar o eixo $y$ mais de uma vez.
  \item As seguintes fórmulas se referem ŕ mesma funçăo:
  \begin{itemize}
    \item $f(x) = x^2$
    \item $f = \{(x,y): x,y \in \Z, y=x^2\}$
    \item $f = \{\dotsc, (-2,4), (-1,1), (0,0), (1,1), (2,4),\dotsc\}$
  \end{itemize}
  \item Se $f: A \mapsto B$, o conjunto $A$ se chama \textbf{domínio}
        e o conjunto $B$ se chama \textbf{imagem}.
  \item As funçőes podem ser:
  \begin{itemize}
    \item \textbf{Um-a-um}, quando cada valor de entrada resulta num valor de saída
          diferente (sempre que $(x,b),(y,b)\in f$, entăo $x=y$).
    \item \textbf{Sobre}, quando em $f: A \mapsto B$, sempre existe um valor em $B$ existe
          um valor em $A$ correspondente.
    \item \textbf{Bijeçăo}, quando a funçăo é ao mesmo tempo um-a-um e sobre.
  \end{itemize}
  \item Em $f: A \mapsto B$, se $|A|>|B|$, entăo $f$ năo é um-a-um. Se $|B|>|A|$, entăo
        $f$ năo é sobre.
  \item Funçőes compostas săo funçőes que săo aplicadas sobre outras funçőes. Exemplo:
        $(g \circ f)(a) = g(f(a))$.
  \item $h \circ (g \circ f) = (h \circ g) \circ f$
\end{itemize}
